\begin{frame}
  \frametitle{Stating the problem}
  How can we add edges to a graph to make the ratio of the closeness centrality of two vertices as close to 1 as possible?\\
  \pause
  \begin{problembox}{Closeness Ratio Improvement}
    \Input& A graph $G=(V,E)$, $a,b\in V$, and $k\in\mathbb{N}$. \\
    \Prob& Find a set of edges $T\subseteq V^2-E$ of size at most $k$ which maximizes the value of $\frac{\min(c_{G+T}(a),c_{G+T}(b))}{\max(c_{G+T}(a),c_{G+T}(b))}$. \\
  \end{problembox}
\end{frame}
